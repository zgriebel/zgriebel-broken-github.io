This project utilized tweepy, the Twitter API which is no longer freely available and then used NLP analysis from a different workbook. This code is a few years old and I did not want to run into more intances with now banned tweepy so I have recently annotated it with notes from my presentation without reorganizaing it.

def clean_data(data,rows):
#making dict of mentioned for other purposes
  mentioned_dict={}

  for i in range(0,rows):
    if i%1000==0:
      print(i)

    tweet=data.loc[i,'tweet_text']
    mentioned=(data.loc[i,'accounts_mentioned'])
    mentioned_list=re.findall('\S+',str(mentioned),)
    name_list=[]



    #cannot set cell to a list
    name_str=''

    blue_check_count=0
    rando_count=0

    for name in mentioned_list:
      #gettng just the handle


      name=name.replace('[','')
      name=name.replace('.','')
      name=name.replace(']','')
      name=name.replace("'",'')
      name=name.replace(':','')
      name=name.replace(',','')
      name=name.replace(' ','')
      name=name.replace('','')
      tweet=tweet.replace(name,'')
      tweet=tweet.replace('@','')

      #making mentioned into a list

      if name =='':
        data.loc[i,'accounts_mentioned']='none'
        mentioned_dict[i]=['none']

      else:


        name_list.append(name)
        name_str+=' '+name
        #makeing dict of each rows mentioned
      mentioned_dict[i]=name_list
      data.loc[i,'accounts_mentioned']=name_str

      tweet=str(tweet)
      link=(re.findall('http\S+',tweet))

      link=(str(link))
      link=link.replace('[','')
      link=link.replace(']','')
      link=link.replace("'",'')
      if link!='':
        data.loc[i,'used_link']=True
      else:
        data.loc[i,'used_link']=False
      #cleaning up the tweet to just give us the content of the meaning full words, taking out the links and meaningless symbols
      tweet=tweet.replace(link,'')
      tweet=tweet.replace('RT','')
      #turning acounts mentioned into a list


      blob=TextBlob(tweet)
      data.loc[i,"polarity"]=float(blob.sentiment.polarity)
  return(data,mentioned_dict)

def flatten_list(_2d_list):
    flat_list = []
    # Iterate through the outer list
    for element in _2d_list:
        if type(element) is list:
            # If the element is of type list, iterate through the sublist
            for item in element:
                flat_list.append(item)
        else:
            flat_list.append(element)
    return flat_list

def get_data(name,handle,data_set):
    list_of_all=[]
    tweets=tweepy.Cursor(api.user_timeline, screen_name=handle, count=200).items(3500)
    for tweet in tweets:
        #tweet id
        tweet_id=tweet.id
        #twet text
        tweet_text= tweet.text
        #time
        time=tweet.created_at
        rt=re.findall('RT\s',tweet_text)
        #is a retweet
        if rt !=[]:
            is_rt=True
        else:
            is_rt=False
            #gambling words used
        gambling_words=re.findall('bet|gamble|gambling|over|under|spread|cover|odds|win|lose|money|parlay|line', tweet_text.lower())
        #about gambling
        if gambling_words !=[]:
            about_gambling=True
        else:
            about_gambling=False
        mention=re.findall('\@(\S+)',tweet_text)
        if mention !=[]:
            if name in mention:
                mention.remove(name)
            #accounts_mentioned
        accounts_mentioned=mention
        about=re.findall('pmt|pardon',tweet_text.lower())
        if about!=[]:
            related_to_pmt=True
        else:
            related_to_pmt=False
        list_for_data=name,handle,tweet_id,tweet_text,time,is_rt,accounts_mentioned,about_gambling,related_to_pmt
        list_of_all.append(list_for_data)
        new_data=pd.DataFrame(list_of_all, columns = ['name','handle','tweet_id','tweet_text','time','is_retweet','accounts_mentioned','is_about_gambling','about_pmt'])


    return(new_data)
